<!-- Add this to your <head> -->
<meta property="og:image"
      content="https://ogsnap.vercel.app/api/og
        ?title=Your+Page+Title
        &subtitle=Your+description
        &template=blog
        &brand=YourSite
        &accent=%236366f1" />

<!-- Dynamic in Next.js, Astro, etc: -->
<meta property="og:image"
      content={`https://ogsnap.vercel.app/api/og?title=${encodeURIComponent(post.title)}&brand=MyBlog`} />
